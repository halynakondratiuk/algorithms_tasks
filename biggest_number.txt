public static String biggestNumber(int[] numbers) {
        StringBuffer someStr = new StringBuffer();
        int size = numbers.length;
        String result = "";
        
        
        for (int i=0; i<size; i++) {
            someStr.append(numbers[i]).append(" ");
        }
        
        String[] strNums = someStr.toString().split(" ");
        String temp = "";
        
        for (int j=0; j<strNums.length-1; j++) {
            int compare = strNums[j+1].compareTo(strNums[j]);
            if ( compare > 0 ) {
                temp = strNums[j];
                strNums[j] = strNums[j+1];
                strNums[j+1] = temp;
            }
            result = result.concat(strNums[j]);
        }
        
        
        // return someStr.toString();
        return result;
    }
